                    phi0 = phi1
    
            end = time.clock()
            delta_t = end-start
            temp_times[0][0] = i
            temp_times[0][1] = delta_t
            times = np.concatenate((times,temp_times),axis=0)
            np.savetxt(file,times,fmt='%.4e')
